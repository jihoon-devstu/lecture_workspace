서블릿은 클래스이므로 , 웹브라우저에서 직접 클래스명을 명시하여 접속할 수 없다.
따라서 , 브라우저에서 클래스에 접근할 수 있도록 , 간접 맵핑을 시도한다.

<servlet>
	<servlet-name>MyServlet</servlet-name>
	<servlet-class>com.sinse.mysite.test.MyName</servlet-class>
</servlet>

<servlet-mapping>
	<servlet-name>MyServlet</servlet-name>
	<url-pattern>/test</url-pattern>
</servlet-mapping>
